LaplacePo<-function(y,E0,V0,tol=1E-5,n=20)
{#MF 04/10/18
  z0<-E0
  d<-1
  for (i in 1:n)
  {if (abs(d)>tol)
  {ez<-exp(z0)
  f<-y-(z0-E0)/V0-ez
  fd<--(1/V0)-ez
  d<-f/fd
  z0<-z0-d
  }
  }
  mean<-z0
  var<- -1/fd
  return(list(E1=mean,V1=var))
}
